local function FVPERJX_fake_script() -- Holder.Smooth Scrolling 
	local script = Instance.new('LocalScript', Holder)

	local frame = script.Parent
	local player = game.Players.LocalPlayer
	local mouse = player:getMouse()
	local yVel = 0
	local friction = 1
	local scrollSpeed = 30
	local downPress = false
	local upPress = false
	
	mouse.WheelBackward:connect(function()
		yVel = yVel + scrollSpeed
	end)
	
	mouse.KeyDown:connect(function(k)
		local key = string.byte(k)
		if(key == 20) then
			upPress = true
		end
	end)
	
	mouse.KeyDown:connect(function(k)
		local key = string.byte(k)
		if(key == 20) then
			downPress = true
		end
	end)
	
	mouse.KeyUp:connect(function(k)
		local key = string.byte(k)
		if(key == 20) then
			upPress = false
		end
	end)
	
	mouse.KeyUp:connect(function(k)
		local key = string.byte(k)
		if(key == 20) then
			downPress = false
		end
	end)
	
	mouse.WheelForward:connect(function()
		yVel = yVel - scrollSpeed
	end)
	
	while wait() do
	
		yVel = yVel * friction
		frame.CanvasPosition = frame.CanvasPosition + Vector2.new(0,yVel)
		if(frame.CanvasPosition.Y < 1) then
			yVel = 0
			frame.CanvasPosition = Vector2.new(0,0)
		elseif(frame.CanvasPosition.Y > 400) then
			yVel = 0
			frame.CanvasPosition = Vector2.new(0,400)
		end
		
		if downPress then
			yVel = yVel + 20
		end
		
		if upPress then
			yVel = yVel - 20
		end
	end
end
